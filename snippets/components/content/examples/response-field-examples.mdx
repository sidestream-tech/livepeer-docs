## CustomResponseField

### Basic Usage

<CustomResponseField name="id" type="string">
  Unique identifier for the resource
</CustomResponseField>

<CustomResponseField name="name" type="string">
  Display name of the resource
</CustomResponseField>

### With Default Value

<CustomResponseField name="status" type="string" default="active">
  Current status of the resource
</CustomResponseField>

### Required Field

<CustomResponseField name="apiKey" type="string" required>
  API key for authentication (required)
</CustomResponseField>

## ValueResponseField

### Basic Value Field

<ValueResponseField 
  name="streamId" 
  type="string"
  description="The unique stream identifier"
  post={["abc123xyz"]}
/>

/>

/>

## ResponseFieldExpandable

### Basic Expandable Group

<ResponseFieldExpandable title="Stream Configuration">
  {[
    {
      name: "name",
      type: "string",
      description: "Stream name",
      post: ["My Live Stream"]
    },
    {
      name: "profiles",
      type: "array",
      description: "Transcoding profiles",
      post: ["[720p, 480p, 360p]"]
    },
    {
      name: "record",
      type: "boolean",
      description: "Enable recording",
      post: ["true"]
    }
  ]}
</ResponseFieldExpandable>

### Nested Configuration

<ResponseFieldExpandable title="Advanced Settings">
  {[
    {
      name: "multistream",
      type: "object",
      description: "Multistream configuration"
    },
    {
      name: "multistream.targets",
      type: "array",
      description: "List of multistream targets"
    },
    {
      name: "webhooks",
      type: "object",
      description: "Webhook configuration"
    },
    {
      name: "webhooks.url",
      type: "string",
      description: "Webhook endpoint URL",
      post: ["https://example.com/webhook"]
    }
  ]}
</ResponseFieldExpandable>

## ResponseFieldAccordion

### Basic Accordion

<ResponseFieldAccordion title="API Response Fields">
  {[
    {
      name: "id",
      type: "string",
      description: "Unique asset identifier",
      post: ["asset_abc123"]
    },
    {
      name: "status",
      type: "string",
      description: "Processing status",
      post: ["ready"]
    },
    {
      name: "createdAt",
      type: "timestamp",
      description: "Creation timestamp",
      post: ["1640000000000"]
    }
  ]}
</ResponseFieldAccordion>
